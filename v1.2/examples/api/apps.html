编写入口 ​

main.ts

ts
import { createApp, APlugin, AEvent } from 'alemonjs'

class word extends APlugin {
  constructor() {
    super({
      rule: [
        {
          reg: /^\/滴滴$/,
          fnc: 'post'
        }
      ]
    })
  }
  async post(e: AEvent) {
    e.reply('哒哒')
  }
}

// 创建应用
const app =createApp(import.meta.url)
  // 使用 word
app.use(word)
  // 挂载
app.mount()
